\section{Introduction}

This document explores the survival of a particle that lives in 1-dimensional space with a timed gate. A particle travels for some period of time repeatedly. At the end of each period, if the gate is off, the particle dies. Otherwise, it continues traveling with the same period.

\section{The Gate–Particle Model}

We normalize the aforementioned period to be 1.

\begin{definition}
A \textit{\textbf{gate}} with \textit{on ratio} $\onRatio\in(0,1)$ is \textit{open} at time $t$ if $t\in \safeSet:=[0,r)$ and \textit{closed} if $t\in \holeSet:=[r,1)$. 
\label{def:gate}
\end{definition}

\begin{definition}
A \textit{\textbf{particle}} with \textit{travel time} $\travelTime\in(0,1)$ and \textit{phase} $\phase\in[0,1)$ has \textit{arrival times}
\[
\arrivalTime{n}=\phase+\travelTime n \ (\modx 1) \ \ \ \forall n\in \mathbb{N}.
\] 
\label{def:particle}
\end{definition}

\begin{definition}
 The \textit{\textbf{fatal index}} of a particle is the first index at which the particle arrives when the gate is closed
\[
\fatalIndex(\travelTime,\phase,\onRatio)=\min\{n : \arrivalTime{n}\in \holeSet\}.
\] 
\label{def:fatal_index}
\end{definition}

\begin{definition}
 The \textit{\textbf{lifespan}} of a particle is defined as
\begin{center}    
$\lifespan(\travelTime,\phase,\onRatio)=\travelTime \cdot \fatalIndex.$\footnote{$\lifespan$ can alternatively be defined, perhaps more intuitively, as $\lifespan=\travelTime \cdot (\fatalIndex+1)$.}
\end{center}
If a particle has no fatal index, we say $\lifespan=\infty$.

\label{def:lifespan}
\end{definition}

\subsection{Optimization}

\begin{definition}
 The set of \textit{\textbf{optimal travel times}} $\optimalTravelTimes$ is the set of travel times that maximize lifespan
\[
\optimalTravelTimes(\phase,\onRatio)=\argmax_{\travelTime\in\mathbb{Q}} \lifespan(\travelTime,\phase,\onRatio).
\] 
In addition, we denote the longest optimal travel time $\longestTravelTime=\max \optimalTravelTimes$.
\label{def:optimal_travel_times}
\end{definition}

\begin{proposition}
If $\travelTime\in \mathbb{R} \setminus \mathbb{Q}$, then $\fatalIndex$ exists.
\begin{proof}
    By Weyl's equidistribution theorem, the orbit $\{\arrivalTime{n}\}$ is equidistributed on $[0,1)$ when $\travelTime\in\mathbb{R} \setminus \mathbb{Q}$. Since $H=[r,1)$ has positive measure $1-r>0$, there exists an $n$ such that $\arrivalTime{n}\in H$ and therefore $\fatalIndex$ exists.
\end{proof}
\end{proposition}

Now it is natural to wonder about the case where $\travelTime\in\mathbb{Q}$. We will explore this with the help of a few Lemmas.

\begin{lemma}  If $\alpha=p/q$ in lowest terms, arrival times are $q$-cyclic:
\[
    \left(\phase + \alpha n\right) \ \modx 1 = \left(\phase + \alpha (n+q)\right) \ \modx 1
\]
\begin{proof}
    \begin{align*}
        \left(\phase + \alpha (n+q)\right) \ \modx 1 = & \left(\phase + \frac{p}{q} (n+q)\right) \ \modx 1\\
        = & \left(\phase + \frac{pn}{q} +p\right) \ \modx 1\\
        = & \left(\phase + \frac{pn}{q}\right) \ \modx 1 = \left(\phase + \alpha n\right) \ \modx 1.
    \end{align*}
\end{proof}
\end{lemma}

\begin{lemma} If $\alpha=p/q$ in lowest terms, arrival times form a regular $q$-gon:
\[
\Biggl\{ \left(\phase + \alpha n\right) \modx 1 : n\in\mathbb{N} \Biggr\} = \Biggl\{ \left(\phase + \frac{k}{q}\right) \modx 1 : k\in\{0,1,...,q-1\}\Biggr\}.
\]
\begin{proof}
Denote
\[
A=\Biggl\{ \left(\phase + \alpha n\right) \modx 1 : n\in\mathbb{N} \Biggr\},B=\Biggl\{ \left(\phase + \frac{k}{q}\right) \modx 1 : k\in\{0,1,...,q-1\}\Biggr\}.
\]
\begin{align*}
    t\in A\implies t=\left(\phase + \frac{pn}{q}\right) \modx 1 
\end{align*}
and by the division algorithm, $pn=qn'+k$ for some $n'\in\mathbb{N},k\in\{0,1,...,q-1\}$ so
\begin{align*}
    t =\left(\phase + \frac{qn'+k}{q}\right) \modx 1 =\left(\phase + n'+\frac{k}{q}\right) \modx 1 =\left(\phase +\frac{k}{q}\right) \modx 1\in B.
\end{align*}
Thus, $A \subseteq B$.\\

\end{proof}
\end{lemma}

\begin{proposition}
If $\travelTime=p/q$ in lowest terms, then $\fatalIndex$ exists if and only if
\[
\phase + \frac{q-1}{q} (\modx 1) \in \holeSet.
\]
\begin{proof}
$(\impliedby)$\\
Notice first that for any general $k\in\mathbb{N}$, there exists $n\in\mathbb{Z}$ and $k'\in\{0,1,...,q-1\}$ such that $k=qn+k'$ and so we have
\begin{align*}
\left(\phase + \frac{k}{q}\right) \modx 1 = \left(\phase + \frac{qn+k'}{q}\right) \modx 1 = & \left(\phase + n + \frac{k'}{q}\right) \modx 1\\
= & \left(\phase + \frac{k'}{q}\right) \modx 1.
\end{align*}
% also not finished
\end{proof}
\end{proposition}
