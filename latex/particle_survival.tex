\section{Introduction}

We study a particle whose successive arrival times at a gate are generated by a circle rotation $R_\alpha: \mathbb{R}/\mathbb{Z} \to \mathbb{R}/\mathbb{Z}$, $x \mapsto x + \alpha \pmod{1}$. The gate is periodically open on an interval $\safeSet\subset\mathbb{R}/\mathbb{Z}$ and closed on its complement $\holeSet$. The particle survives as long as its orbit under $R_\alpha$ avoids $\holeSet$; the first visit to $\holeSet$ is fatal.

This is an instance of an \textit{open dynamical system}---a dynamical system in which orbits escape (or, in our language, die) upon entering a designated hole \cite{PianigianYorke79, DemersYoung06}. The specific setting of circle rotations with holes has been studied by Bunimovich and Yurchenko \cite{BunimovichYurchenko11} and Dettmann \cite{Dettmann12}, among others. The structural properties of circle rotation orbits that govern escape are classical \cite{KatokHasselblatt95}.

Our contributions are the analysis of a \textit{lifespan} objective $\lifespan{\travelTime} = \travelTime \cdot \fatalIndex{\travelTime}$ and further analysis under stochastic scenarios.

\section{Preliminaries: Circle Rotations}

We collect standard definitions and facts about circle rotations that underpin the model.

\begin{definition}
\label{def:rotation}
Rotations and orbits of rotations are defined as follows
\begin{enumerate}
\item Rotation on a circle of unit length by angle $2\pi\travelTime$ ($0<\alpha<1$) with offset $\phase \in (0,1]$ is denoted
\[
R_\travelTime(\phase):=\phase+\travelTime \ \ (\modx 1).
\]
We pick $\alpha\neq0,1$ to avoid degenerate cases in later results.
\item Iterations of a rotation are denoted
\[
R^n_\travelTime(\phase):=\phase+n\travelTime \ \ (\modx 1).
\]
\item The orbit of a rotation is the set of its iterations
\[
\orbit{\travelTime}{\phase}:=\{\rotation{\travelTime}{n}{\phase} : n\in\mathbb{N}\}.
\]
\end{enumerate}
    
\end{definition}

\begin{proposition}[Orbit structure of rational rotations]
If $\travelTime = p/q$ in lowest terms, then
    \[
    \orbit{\travelTime}{\phase} = \left\{\left(\phase + \frac{k}{q}\right) \modx 1 : k \in \{0,1,\ldots,q-1\}\right\},
    \]
    i.e.\ $q$ equally spaced points on the circle. This follows from the fact that $\gcd(p,q)=1$ implies multiplication by $p$ is a bijection on $\mathbb{Z}/q\mathbb{Z}$.
\label{prop:rational_orbit}
\end{proposition}

\begin{corollary}[Maximal orbit]
If $\travelTime=p/q$ in lowest terms, then
\[
\max\orbit{\travelTime}{\phase} = \phase + \frac{\left\lceil q(1-\phase) \right\rceil - 1}{q}.
\]
\begin{proof}
    By Proposition~\ref{prop:rational_orbit}, every arrival time is in the form
    \[
        \left(\phase+\frac{k}{q}\right) \ \modx 1:k\in\{0,1,...,q-1\}.
    \]
    Note that arrival times fit into two categories because $\phase \in [0,1)$:
    \[
        \left(\phase+\frac{k}{q}\right) \ \modx 1=
        \begin{cases} 
        \phase+\frac{k}{q} & \text{if }\phase+\frac{k}{q} < 1 \\
        \phase+\frac{k}{q}-1 & \text{if }\phase+\frac{k}{q} \ge 1.
        \end{cases}
    \]
    From this we can see that $\max\orbit{\travelTime}{\phase}$ always occurs when $\phase+\frac{k}{q} < 1$ because
    \[
    \phase+\frac{k_i}{q} > \phase+\frac{k_j}{q}-1 \  \ \forall k_i,k_j\in\{0,1,...q-1\}.
    \]
    So we want the biggest $k$ such that:
    \[
    \phase + \frac{k}{q} < 1.
    \]
    This is equivalent to $k<q(1-\phase)$, so the smallest $k$ which breaks this inequality is $k=\left\lceil q(1-\phase) \right\rceil$ and the largest $k$ that keeps the inequality is
    \[
    k^*=\left\lceil q(1-\phase) \right\rceil-1
    \]
    and the result follows.
\end{proof}
\label{cor:max_travel_point}
\end{corollary}

\section{The Gate--Particle Model}

\subsection{Definitions}

It is useful to imagine the ordered iterations of a rotation as the steps that a particle takes on the circle with offset $\phase$ and travel time $\travelTime$ for each step. We then introduce a \lq gate\rq \ on the circle that is open for some ratio $r$ of the circle.

\begin{definition}
A \textit{\textbf{gate}} with \textit{ratio} $\onRatio\in(0,1)$ partitions $\mathbb{R}/\mathbb{Z}$ into the \textit{safe set} $\safeSet:=[0,r)$ and the \textit{hole} $\holeSet:=[r,1)$.
\label{def:gate}
\end{definition}

\begin{definition}
The \textit{\textbf{first escape time}} is the first rotation at which the orbit enters the hole:
\[
\fatalIndexFn{\travelTime}{\phase}{\onRatio}:=\min\{n \geq 0: \rotation{\travelTime}{n}{\phase}\in \holeSet\}.
\]
If there is no such $n$, we set $\fatalIndex{\travelTime}:=\infty$.
\label{def:fatal_index}
\end{definition}

\subsection{Existence of a First Escape Time}

\begin{proposition}[Travel time condition for escape time]
\label{prop:travel_time_condition}
If $\travelTime=p/q$ in lowest terms, then $\fatalIndex{\travelTime} = \infty$ if and only if
\[
    \max\orbit{\travelTime}{\phase} < \onRatio.
\]
\begin{proof}
By Def \ref{def:fatal_index}, $\fatalIndex{\travelTime}$ is infinite if and only if  $\forall t \in \orbit{\travelTime}{\phase},t<\onRatio$. \\
It suffices to check only $\orbit{\travelTime}{\phase} < r$.
\end{proof}
\end{proposition}
\newpage
\noindent
We now set out to prove an escape time condition given $\phase$ and $\onRatio$.

\begin{lemma}[Qualifying $\lceil q(1-\phase)\rceil$]
\[
\frac{k}{q}\le\phase<\frac{k+1}{q} \implies \lceil q(1-\phase)\rceil=q-k.
\]
\begin{proof}
\label{lem:qualifying}
\begin{align*}
    \frac{k}{q}\le\phase<\frac{k+1}{q} & \iff -\frac{k+1}{q}<-\phase\le-\frac{k}{q}\\
    & \iff 1-\frac{k+1}{q}<1-\phase\le1-\frac{k}{q}\\
    & \iff q-(k+1)<q(1-\phase)\le q-k\\
    & \iff \lceil q(1-\phase)\rceil=q-k.
\end{align*}
\end{proof}
\end{lemma}

\begin{proposition}[Existence of a travel time yielding an escape time] There exists an $\travelTime$ such that $\fatalIndex{\travelTime} = \infty$ if and only if at least one of the following are true
\begin{enumerate}
    \item $\phase+\frac{1}{2}<\onRatio$
    \item $\frac{2}{3}<\phase+\frac{1}{3}<\onRatio$
    \item $\frac{1}{2}<\phase<\onRatio$.
\end{enumerate}
\begin{proof}
    By Cor \ref{cor:max_travel_point} and Prop \ref{prop:travel_time_condition}, we want to find exactly when there exists an $\travelTime=\frac{p}{q}$ in lowest terms such that
\begin{equation}
\label{eq:max_orbit}
    \max\orbit{\travelTime}{\phase} = \phase + \frac{\left\lceil q(1-\phase) \right\rceil - 1}{q} < \onRatio.
\end{equation}
And by Lem \ref{lem:qualifying}, for all $k\in\mathbb{Z}_q$ we have
\begin{align*}
\frac{k}{q}\le\phase<\frac{k+1}{q} & \iff \lceil q(1-\phase)\rceil=q-k.\\
&\iff \max\orbit{\travelTime}{\phase} = \phase + \frac{q-k - 1}{q} = \phase + 1-\frac{k + 1}{q}
\end{align*}
so Eq \ref{eq:max_orbit} becomes
\[
\phase + 1-\frac{k + 1}{q} < \onRatio.
\]
Thus, we have regions of $(\phase,\onRatio)$ that yield infinite $\fatalIndex{\travelTime}$ for a given $q$:
\[
I_q=\bigcup_{k \in \mathbb{Z}_q}\biggl\{(\phase,\onRatio): \left(\frac{k}{q}\le\phase<\frac{k+1}{q}\right) \land \left(\phase + 1-\frac{k + 1}{q} < \onRatio\right)\biggr\}.
\]
We can then construct the set of all adequate $(\phase,\onRatio)$:
\[
I=\bigcup_{q >1} I_q.
\]
Note that we skip $q=1$ because $0<\travelTime<1$ by Def \ref{def:rotation}.
\\
\\
Now, given $\phase$ and $\onRatio$, there exists an adequate $\travelTime$ if and only if  $(\phase,\onRatio) \in I$. We leave it as an exercise for the reader to show that $I=I_2\cup I_3$ and consequently that the 3 proposed inequalities cover $I$ exactly.
\end{proof}
\end{proposition}

\subsection{Lifespan Optimization}

\begin{definition}
\label{def:lifespan}
The \textit{\textbf{lifespan}} of a particle is how long it can travel before it enters the hole:
\[
\lifespanFn{\travelTime}{\phase}{\onRatio}:=\travelTime \cdot \fatalIndexFn{\travelTime}{\phase}{\onRatio}.\footnote{An alternative convention could be $\lifespan{\travelTime}=\travelTime \cdot (\fatalIndex{\travelTime}+1)$.}
\]
\end{definition}

\begin{definition}
The set of \textit{\textbf{optimal rotations}} is
\[
\optimalTravelTimes(\phase,\onRatio):=\argmax_{\travelTime\in\mathbb{Q}} \lifespanFn{\travelTime}{\phase}{\onRatio}.
\]
We denote the largest optimal rotation by
\[
\longestTravelTime:=\max \optimalTravelTimes.
\]
\label{def:optimal_travel_times}
\end{definition}
